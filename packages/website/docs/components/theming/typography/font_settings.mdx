# Font settings

import { FontFamilyPreview } from './preview_font_settings';

<Example>
  <Example.Description>
    ### `euiTheme.font.family` <Badge color="hollow">token</Badge>

    The whole font family stack for all parts of the UI. We encourage only customizing the first font in the stack.

    All of EUI defaults to the base `font.family`. However, you can use this token to change specific instances.

  </Example.Description>
  <Example.Preview>
    <FontFamilyPreview />
  </Example.Preview>
  <Example.Snippet>
    ```tsx
    css`
      font-family: ${euiTheme.font.family};
    `
    ```
  </Example.Snippet>
</Example>

import { FontDefaultUnitsPreview } from './preview_font_settings';

<Example>
  <Example.Description>
    ### `euiTheme.font.defaultUnits` <Badge color="hollow">token</Badge>

    Sets the default units used for font size & line height set by UI components like EuiText or EuiTitle. Defaults to `rem`.

    Note: This may override by some internal usages, e.g. EuiText's `relative` size which must use `em`.

  </Example.Description>
  <Example.Preview>
    <FontDefaultUnitsPreview />
  </Example.Preview>
  <Example.Snippet>
    ```tsx
    <EuiProvider modify={{ font: { defaultUnits: 'px' } }}>
      <EuiText>
        <p>Hello world</p>
      </EuiText>
    </EuiProvider>
    ```
  </Example.Snippet>
</Example>
