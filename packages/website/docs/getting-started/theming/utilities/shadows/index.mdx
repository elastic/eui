# Shadows

EUI provides 5 shadow sizes (`xs`, `s`, `m`, `l`, `xl`) that can be applied in two directions: `down` (default) and `up`.

The provided EUI shadow utilities handle the following additional style adjustments:

- adds a light, 4-sided `border` style in Dark mode to increase elevated item visibility
- replaces the `box-shadow` in High Contrast mode with `border` to ensure visibility in high contrast themes

## Usage

<Example.Snippet>
  ```ts
  import { euiShadow } from '@elastic/eui';

  const styles = css`
    ${euiShadow(euiThemeContext, 's')};
    ${euiShadow(euiThemeContext, 's', { direction: 'up' })};
  `;  
  ```

### Dark mode floating border customization

By default the shadow utilities add a light, 4-sided `border` style in Dark mode. In some cases where different shadowed elements are being composed together, 
this might result in duplicate neighboring borders. To adjust this, use the `border` option on the `euiShadow` utility to adjust or remove the border output.

:::info The additional border is only applied and visible in dark mode.
:::

<Example.Snippet>
  ```ts
  import { euiShadow } from '@elastic/eui';

  const styles = css`
    ${euiShadow('s', { border: 'none'})}
    ${euiShadow('s', { border: 'all'})}
    ${euiShadow('s', { border: 'top'})}
    ${euiShadow('s', { border: 'bottom'})}
    ${euiShadow('s', { border: 'left'})}
    ${euiShadow('s', { border: 'right'})}
    ${euiShadow('s', { border: 'horizontal'})}
    ${euiShadow('s', { border: 'vertical'})}
  `;  
  ```
</Example.Snippet>

```tsx interactive
import React, { useState } from 'react';
import { useEuiTheme, EuiSelect, EuiSpacer, EuiCodeBlock, EuiPanel, EuiThemeProvider } from '@elastic/eui';

type ShadowBorderOption = 'none' | 'all' | 'top' | 'bottom' | 'left' | 'right' | 'horizontal' | 'vertical';

const Example = ({ selectedBorder }: { selectedBorder: ShadowBorderOption }) => {
  const euiThemeContext = useEuiTheme();
  const { euiTheme } = euiThemeContext;

  const styles = css`
    border-radius: ${euiTheme.border.radius.medium};
    ${euiShadow(euiThemeContext, 'm', { border: selectedBorder })}}
  `;

  return (
    <EuiPanel hasShadow={false} color="plain">
      <EuiCodeBlock css={styles} language="js">
          {`const styles = css\`
  \${euiShadow(euiThemeContext, 'm', { border: '${selectedBorder}' })}
\`}`}
      </EuiCodeBlock>
    </EuiPanel>
  );
}

export default () => {
  const [selectedBorder, setSelectedBorder] = useState<ShadowBorderOption>('all');
  const options = [
    { value: 'none', text: 'none' },
    { value: 'all', text: 'all' },
    { value: 'top' , text: 'top' },
    { value: 'bottom' , text: 'bottom' },
    { value: 'left' , text: 'left' },
    { value: 'right', text: 'right' },
    { value: 'horizontal', text: 'horizontal' },
    { value: 'vertical', text: 'vertical' },
  ];

  return (
    <>
      <EuiThemeProvider colorMode="dark">
        <Example selectedBorder={selectedBorder} />
      </EuiThemeProvider>
      <EuiSpacer />
      <EuiSelect
        aria-label="Shadow Size" 
        prepend="Border" 
        compressed 
        options={options}
        value={selectedBorder}
        onChange={(e) => { setSelectedBorder(e.target.value) }}
      />
    </>
  );
}
```

## Hover states

Each shadowed surface, when interactive, follows the same hover effect rule: it moves up one level. 

- `xs` hovered → `s`
- `s` hovered → `m`
- `m` hovered → `l`
- `l` hovered → `xl`
- `xl` hovered → `hover.xl`

You don't need to remember this rule though. You can instead use the available `euiShadowHover` utility which 
applies the expected hover shadow based on the base size.

```tsx interactive
import React, { useState } from 'react';
import { useEuiTheme, EuiSelect, EuiSpacer, EuiCodeBlock } from '@elastic/eui';

type ShadowSize = 'xs' | 's' | 'm' | 'l' | 'xl';

export default () => {
  const euiThemeContext = useEuiTheme();
  const { euiTheme } = euiThemeContext;
  const [selectedSize, setSelectedSize] = useState<ShadowSize>('m');
  const options = [
    { value: 'xs', text: 'xs' },
    { value: 's' , text: 's' },
    { value: 'm' , text: 'm' },
    { value: 'l' , text: 'l' },
    { value: 'xl', text: 'xl' },
  ];

  const styles = css`
    ${euiShadow(euiThemeContext, selectedSize)}
    border-radius: ${euiTheme.border.radius.medium};
    
    &:hover {
      ${euiShadowHover(euiThemeContext, selectedSize)}}
    }
  `
  const stylesText = `
const styles = css\`
  \${euiShadow(euiThemeContext, '${selectedSize}')}
  &:hover {
    \${euiShadowHover(euiThemeContext, '${selectedSize}')}
  }
\`
  `

  return (
    <>
      <EuiCodeBlock css={styles} language="js">
        {stylesText}
      </EuiCodeBlock>
      <EuiSpacer />
      <EuiSelect
        aria-label="Shadow Size" 
        prepend="Size" 
        compressed 
        options={options}
        value={selectedSize}
        onChange={(e) => { setSelectedSize(e.target.value) }}
      />
    </>
  );
}
```
