## Wrapping

import { WrappingTextTruncateFnPreview } from './preview_utilities';

<Example>
  <Example.Description>
    ### `euiTextTruncate(maxWidth?)` <Badge color="hollow">function</Badge>

    Truncates text at 100% width of its parent and will display an ellipsis.

    This utility accepts a single optional parameter for customizing the maximum width of the truncated text. If not passed, it defaults to `max-width: 100%;`.

    Tip: When truncating text, it is recommended to include the full text within an HTML title attribute or by wrapping the element within an EuiToolTip.

  </Example.Description>
  <Example.Preview>
    <WrappingTextTruncateFnPreview />
  </Example.Preview>
  <Example.Snippet>
    ```tsx
    css`
      ${euiTextTruncate()}
    `
    ```
  </Example.Snippet>
</Example>

<EuiSpacer />

## Color

import { ColorInheritPreview } from './preview_utilities';

<Example>
  <Example.Description>
    ### `.eui-textInheritColor` <Badge color="hollow">className</Badge>

    Forces the component to inherit its text color from its parent.

    For changing the color of your text to on of the named colors, [use **EuiText** or **EuiTextColor**](../../../display/text#coloring-text).

  </Example.Description>
  <Example.Preview>
    <ColorInheritPreview />
  </Example.Preview>
  <Example.Snippet>
    ```tsx
    <EuiTextColor color="danger">
      <EuiCode className="eui-textInheritColor">I am danger code</EuiCode>
    </EuiTextColor>
    ```
  </Example.Snippet>
</Example>
